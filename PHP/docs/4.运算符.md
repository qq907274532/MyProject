##运算符
<?php  
  $xkHeight = 180;  
  $xmHeight = 175;  

  echo '小康身高为', $xkHeight, '厘米', '<br \>';  
  echo '小明身高为', $xmHeight, '厘米', '<br \>';  
  echo '小康+小明=', $xkHeight+$xmHeight, '厘米', '<br \>';  
  echo '小康-小明=', $xkHeight-$xmHeight, '厘米', '<br \>';  
  echo '小康*小明=', $xkHeight*$xmHeight, '厘米', '<br \>';  
  echo '小康/小明=', $xkHeight/$xmHeight, '厘米', '<br \>';  
  echo '小康%小明=', $xkHeight%$xmHeight, '厘米', '<br \>';  
  
  if($xkHeight > $xmHeight)  
  {  
    echo '小康比小明高', '<br \>';  
  }  
  else   
  {  
    echo '小明比小康高', '<br \>';  
  }  
?>
